## 4. Pipeline de Señales (Diario)

```python
# scripts/daily_update.py — esqueleto conceptual

def daily_pipeline():
    # 1. Descargar / actualizar datos
    update_all_series()
    
    # 2. Health check
    health = run_health_checks()
    if health["pass_rate"] < 0.7:
        alert("Health check failure", health)
        return
    
    # 3. Calcular z-scores rolling 10Y
    zscores = calculate_zscores(window_years=10)
    
    # 4. Cargar β actuales (config)
    betas_endo = load_betas("config/betas_capa1.yaml")
    betas_exo = load_betas_exo()
    
    # 5. Calcular World View
    wv_state = calculate_world_view(zscores)
    log("WorldView", wv_state)
    
    # 6. Calcular Score_Endo y Score_Exo por país
    score_endo = compute_score_endo(zscores, betas_endo, wv_state)
    score_exo = compute_score_exo(zscores, betas_exo)
    
    # 7. Combinar
    score_total = combine_scores(score_endo, score_exo, w_exo_table)
    
    # 8. Generar Señal_FX por par
    pairs = generate_pairs_g10()
    signals = compute_signals(score_total, pairs)
    
    # 9. Filtrar por convicción
    actionable = filter_by_conviction(signals, min_sigma=1.0)
    
    # 10. Aplicar World View veto (§7.4)
    actionable = apply_wv_veto(actionable, wv_state)
    
    # 11. Para cada señal vigente, evaluar timing
    for signal in actionable:
        timing = evaluate_timing(signal)
        if timing.action == "trade-now":
            generate_trade_proposal(signal, timing)
    
    # 12. Update dashboard
    refresh_dashboard()
    
    # 13. Alertas
    check_alerts(signals, wv_state, current_positions)
```
